### function to showcase quicks ###
func bruhShowcase myArgument1 myArgument2 { # return myArgument1 times myArgument2. myArgument1 and myArgument2 are RAM variables
    quick 1 myQuickVariable = myArgument1
    quick 2 myOtherQuickVariable = myArgument2
    # quick variables are stored in regs but are assigned by the programmer, to optimize code

    quick 3 result = myQuickVariable * myOtherQuickVariable # makes it just 1 line in URCL because of quicks being in registers

    return result
}

### function to showcase freeing of stuff
func bruhShowcase2 arg1 arg2 {
    quick quick1 = arg1
    quick quick2 = arg2

    quick quick3 = quick1 * quick2
    free quick1

    quick quick4 = quick2 * 5

    quick quick5 = quick3 + quick4
    return quick5
    # all variables get automatically freed up when ending a variable
}
